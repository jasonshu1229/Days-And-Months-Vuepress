### 作用域提升面试题

```JS
// 题1
var n = 100;

function foo() {
  n = 200;
}
foo();

console.log(n); // 200 
```

```JS
// 题2
function foo() {
  console.log(n); // undefined
  var n = 200;
  console.log(n); // 200
}

var n = 100;
foo();
```

```JS
// 题3 注意这道题：
function foo() {
  var a = b = 100;
  // 转化成下面的代码
  // var a = 10;
  // b = 10;
}

foo();
console.log(a); // a is not defined
console.log(b); // 10
```

```JS
// 题4
var a = 100;

function foo() {
  console.log(a); // undefined
  return
  var a = 100;
}

foo();
```

在函数foo执行上下文内仍能找到 a 的变量，即使在 a 在 return 后面定义。
